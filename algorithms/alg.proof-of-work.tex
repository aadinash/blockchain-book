\begin{algorithm}[H]
  \caption{\label{alg.proof-of-work} The proof-of-work algorithm.}
  \begin{algorithmic}[1]
    \Function{\textsf{PoW}$_{H,T}$}{}
      \State{$\ctr \getsrandomly \{0, 1\}^\kappa$}
      \While{$\true$}
        \Let{B}{\ctr}
        \If{$H(B) \leq T$}
          \State\Return{B}
        \EndIf
        \Let{\ctr}{\ctr + 1}
      \EndWhile
    \EndFunction
  \end{algorithmic}
\end{algorithm}
