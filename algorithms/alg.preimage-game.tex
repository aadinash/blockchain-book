\begin{figure}[t]
\begin{algorithm}[H]
    \caption{\label{alg.preimage-game} The preimage-finding
             game for a hash function $H$.}
    \begin{algorithmic}[1]
        \Function{\sf preimage-game$_{H,\mathcal{A}}$}{$\kappa$}
            \State{$x \getsrandomly \{0, 1\}^\kappa$}
            \Let{y}{H_\kappa(x)}
            \Let{x'}{\mathcal{A}(y)}
            \State\Return $H_\kappa(x') = y$
        \EndFunction
        \vskip8pt
    \end{algorithmic}
\end{algorithm}
\end{figure}
